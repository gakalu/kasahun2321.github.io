*/
function maxOfThree(x,y,z)
{
    let max=0;
    if (x>y && x>z)
    {
        max=x;
    }
    else if(x>y && x<z)
    {
        max=z;
    }
    else if(y>x && y>z)
    {
        max=y;
    }
    else
    {
        max=z;
    }
    return max;
}

console.log("the maximum of three number is:"+maxOfThree(4,5,6));
